# Copy the array from source (0-9) to destination (10-19)
0: LI R1, 0;  # R1 is the read pointer
1: LD R2, R1; # Loading first element
2: LI R3, 1;  # How much we are incrementing each time
3: LI R7, 20; # address of setting new maximum
4: LI R8, 7;  # come back address
5: LI R9, 22;  # ENDING ADDRESS
6: LI R10, 5;  # Number of elements
7: LI R11 255;
8: LI R12 50;
9: LI R13 12;
10: LD R14, R2;
#starts the loop
11: ADD R4, R4, R3;  # R4 is the counter that is incremented
12: JEQ R9, R4, R10;  # R4 is the counter that is incremented
13: LD R5 , R4;  # Loading element from memory into register 5
14: JLT R7, R2, R5;  # Done copy address
15: ADD R14, R14, R5;
16: JLT R12, R5, R11;
17: JR R8;



# max:
20: ADD R2, R5, R0; # setting new maximum
21: JR R13; # jumps back
